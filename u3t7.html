Algoritmo de Ordenamiento Quicksort

Descripción:
El quicksort es un algoritmo eficiente que usa recursión para dividir el arreglo en subarreglos y ordenar cada uno de manera independiente.

Ejemplo:

public class OrdenamientoQuicksort {
public static void quicksort(int[] arreglo, int inicio, int fin) {
if (inicio < fin) {
int indiceParticion = particion(arreglo, inicio, fin);
quicksort(arreglo, inicio, indiceParticion - 1);
quicksort(arreglo, indiceParticion + 1, fin);
}
}

private static int particion(int[] arreglo, int inicio, int fin) {
int pivote = arreglo[fin];
int i = inicio - 1;

for (int j = inicio; j < fin; j++) {
if (arreglo[j] <= pivote) {
i++;
int temp = arreglo[i];
arreglo[i] = arreglo[j];
arreglo[j] = temp;
}
}

int temp = arreglo[i + 1];
arreglo[i + 1] = arreglo[fin];
arreglo[fin] = temp;

return i + 1;
}

public static void main(String[] args) {
int[] numeros = {5, 3, 8, 1, 2};
quicksort(numeros, 0, numeros.length - 1);

System.out.print("Arreglo ordenado: ");
for (int numero : numeros) {
System.out.print(numero + " ");
}
}
}
